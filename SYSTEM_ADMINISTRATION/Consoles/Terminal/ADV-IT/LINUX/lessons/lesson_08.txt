
locate , plocate, whereis
	эти команды ищут только системные файлы

find
	поиск по всем файлам и директориям

find /home -name *.txt
or
find /home -name "*.txt"
	поиск всех файлов с расширением .txt в директории /home


wc file1.txt
wc Dir1/file2.txt
	подсчет количества строк, слов, символов в файле
wc -l file1.txt
	подсчет количества строк в файле
wc -w file1.txt
	подсчет количества слов в файле
wc -c file1.txt
	подсчет количества символов в файле


sort file1.txt
	сортирует содержимое файла по алфавиту не изменяя исходный файла
cat file1.txt
Dima
Andrii

sort file1.txt
Andrii
Dima

sort -n numbers.txt
	Сортирует файл по цифрам а не по символам

cut -d ':' -f 1 file1.txt
	отсекает строки по символу ':' и выводит первый столбец

cut -d ">" -f 3 file3.txt
	отсекает строки по символу ">" и выводит третий столбец

cut -d ':' -f 1,2 file1.txt
	отсекает строки по символу ':' и выводит первый и второй столбец

cut -d ":" -f 1-3 file1.txt
	отсекает строки по символу ':' и выводит первый, второй и третий столбец

cut -d ":" -f 2- file1.txt
	отсекает строки по символу ':' и выводит второй и все последующие столбцы

cut -d ":" -f 2 file4.txt | sort | uniq -c
	отсекает строки по символу ':' и выводит второй столбец, сортирует и удаляет дубликаты
