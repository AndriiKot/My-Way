
touch some_file.txt
	создаёт новый пустой файл
    если приминить touch к уже сушествуещему файлу то содержимое файла останится прижним
	измениться только дата обновления файла
    touch создаёт только файлы, но если ввести touch и имя уже существуещей директории
 	то будет то содежримое директории не изменится но будет произведенно обновление

	открыть файл для просмотра или редоктирования удобно искать 
	в файле ключевые слова а также 'less' редактор предоставляет
	другие полезные функции

cp  some_file.txt some_folder
	копироваине 'some_file.txt' в папку 'some_filder'
    cp some_file ~/Desktop/ скопировать файл на рабочий стол

cp hello*.* some_folder
	скопирует все файлы которые начинаются на 'hello ...' и слюбым расширением

cp hello?.txt some_folder
	скоприрует все файлы у которых начало файла начинается с 'hello' а затем идет
	только один символ любой

cp -v hello*.* some_folder
	визуальное отоброжение в консоли куда были скопированы файлы2

cp -R Dir1 newFolder
	копирование все папки 'Dir1' со все содержимым в папку 'newFolder'

rm Dir1/*.*
	удалить все файлы в папку 'Dir1'
    rm *.* без параметров удалит все файлы в текущей директории
	rm *  удалит все файлы и директории в текущем каталоге

ls -l Dir2
	покажет содержимое каталога 'Dir2' если он находиться в текущей директории

mv 
	перенести файл или переименовать его
mv some_file.txt .some_file.txt
	сделает файл скрытым (просто добавить точку в названия файла)
