
touch some_file.txt
	создаёт новый пустой файл
    если применить touch к уже существующему файлу то содержимое файла останется прежним
	измениться только дата обновления файла
    touch создаёт только файлы, но если ввести touch и имя уже существующей директории
 	то будет то содержимое директории не изменится но будет произведено обновление

cat > new_file.txt
 enter some text
Ctrl + D
		запишет текст в файл 'new_file.txt'

cat new_file.txt
	посмотреть содержимое файла
more new_file.txt
	посмотреть содержимое файла по странично

less new_file.txt
	открыть файл для просмотра или редактирования удобно искать
	в файле ключевые слова а также 'less' редактор предоставляет
	другие полезные функции

cp  some_file.txt some_folder
	копирование 'some_file.txt' в папку 'some_folder'
    cp some_file ~/Desktop/ скопировать файл на рабочий стол

cp hello*.* some_folder
	скопирует все файлы которые начинаются на 'hello ...' и с любым расширением

cp hello?.txt some_folder
	скопирует все файлы у которых начало файла начинается с 'hello' а затем идет
	только один символ любой

cp -v hello*.* some_folder
	визуальное отображение в консоли куда были скопированы файлы2

cp -R Dir1 newFolder
	копирование все папки 'Dir1' со все содержимым в папку 'newFolder'

rm Dir1/*.*
	удалить все файлы в папку 'Dir1'
    rm *.* без параметров удалит все файлы в текущей директории
	rm *  удалит все файлы и директории в текущем каталоге

ls -l Dir2
	покажет содержимое каталога 'Dir2' если он находиться в текущей директории

mv
	перенести файл или переименовать его
mv some_file.txt .some_file.txt
	сделает файл скрытым (просто добавить точку в названия файла)
