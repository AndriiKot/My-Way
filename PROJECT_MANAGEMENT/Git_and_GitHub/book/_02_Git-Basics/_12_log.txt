
git log
	Выводит все коммиты сверху самый последний и дальше по убыванию
  (контрольная сумма SHA-1, имя автора, дата создание , эл. почта)

git log -p 
git log --patch
  Показвает все коммиты с их точными изменениями

git log -p --stat
  Показывает сколько строчек было измененно  (спискок количество изменений)

git log --pretty=oneline
  Выводит коммиты в одну строчку

git log --pretty=format:"%h - %an, %ar : %s"

git log --pretty=format:"%h %s" --graph
