SELECT title, film_id, rating
FROM film f
ORDER BY length DESC;

-- Сортировка по убыванию не выбранного поля в SELECT


SELECT title, film_id, rating
FROM film f
ORDER BY length DESC, title;

-- Комбинированная Сортировка
-- Сначала данные сортируются по убыванию 'length' 
-- а затем по возрастанию поля 'title'
