SELECT title
FROM film
LIMIT 5
OFFSET 10;

!!!

!!! Если после OFFSET 10 то есть после 11 строки 
    то ничего не вернётся 
!!!

!!! Если после OFFSET идёт меньше строк чем указанно
    в LIMIT то вернутся те которые доступны
