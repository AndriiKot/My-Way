-rw-rw-r--               это права доступа

drwxr-xr-x  2 root   root              4096 Nov 21 10:27 apt
d - это папка
rwx - права собственника 
root - собственник
r - read можно читать
w - write можно записывать
x - можно запускать(можно зайти в папку и посмотреть содержимое)
r-x - права доступа группы собственника
root - группа собственника(вторая колонка)
r - read можно читать
-  - нельзя записывать
x  - можете запускать(можно заходить внутрь папки)

ls -lt /var/log          сортировка по дате изменения (сначала самые свежие изменения)
ls -l -S /var/log        сортировка по размеру файла (сначала самые ёмкие файлы)
ls -lSr /var/log         сортировка по размеру файла (сначала самые мелкие файлы)
